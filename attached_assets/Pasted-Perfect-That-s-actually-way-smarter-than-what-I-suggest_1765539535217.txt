## Implementation Logic

### 1. Load and Index Services

```typescript
// Load me-nexus-services.json
const allServices = loadMENextusServices();

// Build L1 index
const l1Map = new Map<string, {
  name: string;
  services: Service[];
  count: number;
}>();

allServices.forEach(service => {
  const l1 = service.hierarchy.l1;
  if (!l1) return;
  
  if (!l1Map.has(l1)) {
    l1Map.set(l1, { name: l1, services: [], count: 0 });
  }
  
  l1Map.get(l1).services.push(service);
  l1Map.get(l1).count++;
});

// Sort by service count (descending)
const l1Categories = Array.from(l1Map.values())
  .sort((a, b) => b.count - a.count);
```

### 2. Cascade Selection

```typescript
// When L1 selected
function onL1Selected(l1Name: string) {
  const category = l1Map.get(l1Name);
  
  // Build hierarchical service list
  const services = buildServiceTree(category.services);
  
  // Update L2/L3 dropdown
  setAvailableServices(services);
  
  // Clear previous L2/L3 selection
  setSelectedService(null);
  
  // Auto-populate TaskSC
  setTaskSC({
    structuralType: `menexus:${sanitize(l1Name)}`,
    structuralProperties: {
      meNexusL1: l1Name,
      serviceCount: category.count
    }
  });
}

// When L2/L3 selected
function onServiceSelected(service: Service) {
  // Build full path for display
  const path = buildPath(service);
  
  // Auto-populate TaskFC
  setTaskFC({
    functionalType: `menexus:${sanitize(service.serviceName)}`,
    functionalProperties: {
      meNexusService: {
        serviceId: service.serviceId,
        serviceName: service.serviceName,
        fullPath: path,
        l1: service.hierarchy.l1,
        l2: service.hierarchy.l2,
        l3: service.hierarchy.l3,
        description: service.description
      },
      omcEquivalent: mapToOMC(service)
    }
  });
}
```

### 3. Build Hierarchical Service Tree

```typescript
function buildServiceTree(services: Service[]): TreeNode[] {
  // Group by L2
  const l2Groups = new Map<string, Service[]>();
  
  services.forEach(service => {
    const l2 = service.hierarchy.l2 || service.serviceName;
    
    if (!l2Groups.has(l2)) {
      l2Groups.set(l2, []);
    }
    
    l2Groups.get(l2).push(service);
  });
  
  // Build tree
  const tree: TreeNode[] = [];
  
  l2Groups.forEach((serviceList, l2Name) => {
    // Check if this L2 has L3 children
    const hasL3 = serviceList.some(s => s.hierarchy.l3);
    
    if (hasL3) {
      // Create L2 parent with L3 children
      const l2Node = {
        label: l2Name,
        value: null, // L2 is not selectable if it has L3
        children: serviceList
          .filter(s => s.hierarchy.l3)
          .map(s => ({
            label: s.hierarchy.l3,
            value: s.serviceId,
            service: s
          }))
      };
      tree.push(l2Node);
    } else {
      // L2 is directly selectable
      const service = serviceList[0];
      tree.push({
        label: l2Name,
        value: service.serviceId,
        service: service
      });
    }
  });
  
  return tree;
}
```

### 4. Map to OMC Equivalent

```typescript
function mapToOMC(service: Service): string {
  const l1 = service.hierarchy.l1;
  const l2 = service.hierarchy.l2;
  
  // Map based on L1/L2
  switch (l1) {
    case 'Animation':
    case 'Compositing':
    case 'Asset Creation':
    case 'FX-Simulation':
    case 'Lighting':
    case 'Rendering':
    case 'Layout':
      return 'omc:CreateVisualEffects';
    
    case 'Production Services':
      if (l2 === 'Pre-Production Services') {
        return 'omc:DevelopCreativeStyle';
      } else if (l2 === 'Principal Photography') {
        return 'omc:Shoot';
      } else if (l2 === 'Post-Production Services') {
        return 'omc:ConformFinish';
      }
      return 'omc:Task';
    
    case 'Editorial':
      return 'omc:Edit';
    
    case 'Virtual Production':
      return 'omc:Shoot';
    
    case 'Game Development':
      return 'omc:CreateVisualEffects';
    
    default:
      return 'omc:Task';
  }
}
```
